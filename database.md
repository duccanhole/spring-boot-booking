- USER table
```
CREATE TABLE USERS (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    name NVARCHAR(255) NOT NULL,
    email NVARCHAR(255) UNIQUE NOT NULL,
    phone NVARCHAR(20) UNIQUE,
    password NVARCHAR(255) NOT NULL,
    role NVARCHAR(50) NOT NULL
);
```
- BUSES table
```
CREATE TABLE BUSES (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    license_plate NVARCHAR(50) UNIQUE NOT NULL,
    seat_capacity INT NOT NULL,
    type NVARCHAR(100) NOT NULL,
    status NVARCHAR(50) NOT NULL
);
```
- DRIVERS table
```
CREATE TABLE DRIVERS (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    user_id UNIQUEIDENTIFIER NOT NULL,
    license_number NVARCHAR(50) UNIQUE NOT NULL,
    status NVARCHAR(50) NOT NULL,
    FOREIGN KEY (user_id) REFERENCES USERS(id) ON DELETE CASCADE
);
```
- ROUTES table
```
CREATE TABLE ROUTES (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    departure NVARCHAR(255) NOT NULL,
    arrival NVARCHAR(255) NOT NULL,
    time_estimate TIME NOT NULL
);
```
- SCHEDULES table
```
CREATE TABLE SCHEDULES (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    bus_id UNIQUEIDENTIFIER NOT NULL,
    driver_id UNIQUEIDENTIFIER NOT NULL,
    route_id UNIQUEIDENTIFIER NOT NULL,
    departure_time DATETIME NOT NULL,
    price DECIMAL(10,2) NOT NULL,

    FOREIGN KEY (bus_id) REFERENCES BUSES(id) ON DELETE CASCADE,
    FOREIGN KEY (driver_id) REFERENCES DRIVERS(id) ON DELETE CASCADE,
    FOREIGN KEY (route_id) REFERENCES ROUTES(id) ON DELETE CASCADE
);
```
- SEATS table
```
CREATE TABLE SEATS (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    schedule_id UNIQUEIDENTIFIER NOT NULL,
    seat_number NVARCHAR(10) NOT NULL,

    FOREIGN KEY (schedule_id) REFERENCES SCHEDULES(id) ON DELETE CASCADE
);
```
- BOOKINGS table
```
CREATE TABLE BOOKINGS (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    user_id UNIQUEIDENTIFIER NOT NULL,
    seat_id UNIQUEIDENTIFIER NOT NULL,
    status NVARCHAR(50) NOT NULL,

    FOREIGN KEY (user_id) REFERENCES USERS(id) ON DELETE CASCADE,  -- Keep cascade for user
    FOREIGN KEY (seat_id) REFERENCES SEATS(id) ON DELETE NO ACTION  -- Remove cascade
);
```
